// /app/api/cliniko/appointments/route.ts
import { clinikoClient } from '@/lib/cliniko';
import { auth } from '@clerk/nextjs';

export async function GET(request: Request) {
  const { userId } = auth();
  if (!userId) return Response.json({ error: 'Unauthorized' }, { status: 401 });
  
  const { searchParams } = new URL(request.url);
  const practitionerId = searchParams.get('practitioner_id');
  const date = searchParams.get('date');
  
  try {
    const appointments = await clinikoClient.getAppointments({
      practitioner_id: practitionerId,
      starts_at: date,
      per_page: 100
    });
    
    // Cache in Neon DB for performance
    await cacheAppointments(appointments);
    
    return Response.json(appointments);
  } catch (error) {
    console.error('Cliniko API Error:', error);
    return Response.json({ error: 'Failed to fetch appointments' }, { status: 500 });
  }
}

export async function POST(request: Request) {
  const { userId } = auth();
  if (!userId) return Response.json({ error: 'Unauthorized' }, { status: 401 });
  
  const appointmentData = await request.json();
  
  try {
    // Create appointment in Cliniko
    const newAppointment = await clinikoClient.createAppointment(appointmentData);
    
    // Log action for compliance
    await auditLog.create({
      user_id: userId,
      action: 'CREATE_APPOINTMENT',
      resource_id: newAppointment.id,
      timestamp: new Date()
    });
    
    return Response.json(newAppointment);
  } catch (error) {
    console.error('Failed to create appointment:', error);
    return Response.json({ error: 'Failed to create appointment' }, { status: 500 });
  }
}

// /lib/google-maps.ts
export class GoogleMapsService {
  private apiKey: string;
  
  constructor() {
    this.apiKey = process.env.GOOGLE_MAPS_API_KEY!;
  }
  
  async geocodeAddress(address: string): Promise {
    // Check cache first
    const cached = await getCachedGeocode(address);
    if (cached) return cached;
    
    const response = await fetch(
      `https://maps.googleapis.com/maps/api/geocode/json?` +
      `address=${encodeURIComponent(address)}&` +
      `region=au&components=country:AU&` +
      `key=${this.apiKey}`
    );
    
    const data = await response.json();
    
    // Cache for 24 hours
    await cacheGeocode(address, data, 86400);
    
    return data;
  }
  
  async calculateTravelTime(origin: string, destination: string): Promise {
    const response = await fetch(
      `https://maps.googleapis.com/maps/api/distancematrix/json?` +
      `origins=${encodeURIComponent(origin)}&` +
      `destinations=${encodeURIComponent(destination)}&` +
      `mode=driving&traffic_model=best_guess&` +
      `departure_time=now&key=${this.apiKey}`
    );
    
    return response.json();
  }
}
